    <div class="badge" style="margin-bottom:10px;"><i></i> تنبيه اللعبة</div>
    <h2 id="overlayTitle">جاهز؟</h2>
    <p id="overlayText">سيظهر الدور الحالي الآن. أعطِ الجهاز للاعب المطلوب ثم اكشف السر.</p>
    <div class="btns" style="justify-content:center;">
      <button class="btn-good" id="overlayOkBtn">حسنًا</button>
    </div>
  </div>
</div>

<script>
  const categories = {
    "أطعمة": ["بيتزا", "برجر", "مكرونة", "فراخ", "كشري", "شاورما", "فلافل", "سمك", "آيس كريم", "شوكولاتة", "فراولة", "عصير"],
    "أماكن": ["مدرسة", "مستشفى", "مطار", "بحر", "صحراء", "حديقة", "متحف", "مكتبة", "سينما", "ملعب", "فندق", "سوق"],
    "حيوانات": ["أسد", "نمر", "فيل", "زرافة", "دلفين", "نسر", "قطة", "كلب", "حصان", "جمل", "دب", "أرنب"],
    "مدن": ["القاهرة", "الإسكندرية", "الجيزة", "أسوان", "الأقصر", "دمشق", "بغداد", "الرياض", "دبي", "الرباط", "تونس", "الخرطوم"],
    "أشياء": ["هاتف", "ساعة", "كتاب", "قلم", "كرسي", "طاولة", "مفتاح", "حقيبة", "مرآة", "مروحة", "مصباح", "نافذة"],
    "مهن": ["طبيب", "معلم", "مهندس", "مصور", "طيار", "طباخ", "نجار", "ممرض", "شرطي", "مزارع", "مقاول", "رسام"]
  };

  const $ = (id) => document.getElementById(id);

  const playersCount = $("playersCount");
  const categorySelect = $("categorySelect");
  const playersList = $("playersList");
  const setupInfo = $("setupInfo");

  const setupScreen = $("setupScreen");
  const roundScreen = $("roundScreen");
  const endScreen = $("endScreen");

  const chipPlayer = $("chipPlayer");
  const chipTurn = $("chipTurn");
  const chipCategory = $("chipCategory");
  const currentPlayerName = $("currentPlayerName");
  const roundHint = $("roundHint");
  const progressBar = $("progressBar");
  const roundCard = $("roundCard");

  const secretStage = $("secretStage");
  const frontText = $("frontText");
  const secretResult = $("secretResult");
  const roundMessage = $("roundMessage");

  const finalWord = $("finalWord");

  const overlay = $("overlay");
  const overlayTitle = $("overlayTitle");
  const overlayText = $("overlayText");
  const overlayOkBtn = $("overlayOkBtn");

  const startBtn = $("startBtn");
  const fillNamesBtn = $("fillNamesBtn");
  const showSecretBtn = $("showSecretBtn");
  const hideSecretBtn = $("hideSecretBtn");
  const nextPlayerBtn = $("nextPlayerBtn");
  const finishGameBtn = $("finishGameBtn");
  const restartBtn = $("restartBtn");
  const newRoundBtn = $("newRoundBtn");
  const backToSetupBtn = $("backToSetupBtn");

  let state = {
    players: [],
    category: "",
    secretWord: "",
    spyIndex: 0,
    currentTurn: 0,
    revealed: false,
    started: false
  };

  function randItem(arr) {
    return arr[Math.floor(Math.random() * arr.length)];
  }

  function shuffle(arr) {
    return [...arr].sort(() => Math.random() - 0.5);
  }

  function showScreen(which) {
    [setupScreen, roundScreen, endScreen].forEach(s => s.classList.remove("active"));
    which.classList.add("active");
  }

  function openOverlay(title, text) {
    overlayTitle.textContent = title;
    overlayText.textContent = text;
    overlay.classList.add("show");
  }

  function closeOverlay() {
    overlay.classList.remove("show");
  }

  function buildCategoryOptions() {
    categorySelect.innerHTML = "";
    Object.keys(categories).forEach(name => {
      const opt = document.createElement("option");
      opt.value = name;
      opt.textContent = name;
      categorySelect.appendChild(opt);
    });
    categorySelect.value = Object.keys(categories)[0];
  }

  function renderPlayersInputs() {
    const count = Number(playersCount.value);
    playersList.innerHTML = "";
    for (let i = 0; i < count; i++) {
      const row = document.createElement("div");
      row.className = "playerItem";
      row.innerHTML = `
        <div class="idx">${i + 1}</div>
        <input type="text" placeholder="اسم اللاعب ${i + 1}" id="player_${i}" />
      `;
      playersList.appendChild(row);
    }
    setupInfo.textContent = `سيتم تجهيز ${count} لاعبين لهذه الجولة.`;
  }

  function fillDemoNames() {
    const names = ["عمر", "أحمد", "سارة", "يوسف", "مريم", "زياد", "نور", "ليان", "كريم", "ملك"];
    [...playersList.querySelectorAll("input")].forEach((input, i) => {
      input.value = names[i] || `لاعب ${i + 1}`;
    });
  }

  function getPlayersFromInputs() {
    const inputs = [...playersList.querySelectorAll("input")];
    const names = inputs.map((inp, i) => inp.value.trim() || `لاعب ${i + 1}`);
    return names;
  }

  function startGame() {
    const players = getPlayersFromInputs();
    if (players.length < 3) {
      openOverlay("عدد قليل", "لازم يكون عندك 3 لاعبين على الأقل.");
      return;
    }

    const category = categorySelect.value;
    const word = randItem(categories[category]);
    const spyIndex = Math.floor(Math.random() * players.length);

    state = {
      players,
      category,
      secretWord: word,
      spyIndex,
      currentTurn: 0,
      revealed: false,
      started: true
    };

    updateRoundUI();
    showScreen(roundScreen);

    openOverlay(
      "بداية الجولة",
      "مرّر الجهاز بين اللاعبين واحدًا واحدًا. كل لاعب يضغط كشف السر ثم يخفيه قبل تسليم الجهاز."
    );
  }

  function updateRoundUI() {
    const total = state.players.length;
    const currentPlayer = state.players[state.currentTurn];

    chipPlayer.textContent = `اللاعب ${state.currentTurn + 1} / ${total}`;
    chipTurn.textContent = `الاسم: ${currentPlayer}`;
    chipCategory.textContent = `الفئة: ${state.category}`;

    currentPlayerName.textContent = currentPlayer;
    roundHint.textContent = `الجولة الحالية تضم ${total} لاعبين. تأكد أن اللاعب الحالي وحده يرى الشاشة.`;
    progressBar.style.width = `${(state.currentTurn / total) * 100}%`;

    secretStage.classList.remove("revealed");
    state.revealed = false;

    frontText.textContent = "اضغط كشف السر";
    secretResult.textContent = state.currentTurn === state.spyIndex ? "أنت الجاسوس 🤫" : state.secretWord;

    roundMessage.textContent =
      state.currentTurn === state.spyIndex
        ? "هذا اللاعب هو الجاسوس. لا تدعه يتكلم كثيرًا 😄"
        : "هذا اللاعب يشاهد الكلمة السرية فقط.";

    nextPlayerBtn.textContent = state.currentTurn === state.players.length - 1 ? "إنهاء الجولة" : "اللاعب التالي";
  }

  function revealSecret() {
    if (!state.started) return;
    secretStage.classList.add("revealed");
    state.revealed = true;
    roundCard.classList.add("shake");
    setTimeout(() => roundCard.classList.remove("shake"), 500);
  }

  function hideSecret() {
    secretStage.classList.remove("revealed");
    state.revealed = false;
  }

  function nextPlayer() {
    if (!state.started) return;

    if (state.currentTurn < state.players.length - 1) {
      state.currentTurn++;
      updateRoundUI();
      openOverlay(
        "دور جديد",
        "سلّم الجهاز للاعب التالي ثم اضغط حسنًا، وبعدها أكشف له الدور."
      );
    } else {
      finishRound();
    }
  }

  function finishRound() {
    progressBar.style.width = "100%";
    finalWord.textContent = `${state.secretWord} — الجاسوس كان: ${state.players[state.spyIndex]}`;

    roundScreen.classList.remove("active");
    endScreen.classList.add("active");

    openOverlay(
      "انتهت الجولة",
      "يمكنك الآن بدء جولة جديدة أو الرجوع للإعدادات لتغيير اللاعبين والفئة."
    );
  }

  function goBackToSetup() {
    state.started = false;
    showScreen(setupScreen);
  }

  playersCount.addEventListener("change", renderPlayersInputs);
  fillNamesBtn.addEventListener("click", fillDemoNames);
  startBtn.addEventListener("click", startGame);
  showSecretBtn.addEventListener("click", revealSecret);
  hideSecretBtn.addEventListener("click", hideSecret);
  nextPlayerBtn.addEventListener("click", nextPlayer);
  finishGameBtn.addEventListener("click", finishRound);
  restartBtn.addEventListener("click", goBackToSetup);
  newRoundBtn.addEventListener("click", goBackToSetup);
  backToSetupBtn.addEventListener("click", goBackToSetup);

  overlayOkBtn.addEventListener("click", closeOverlay);
  overlay.addEventListener("click", (e) => {
    if (e.target === overlay) closeOverlay();
  });

  buildCategoryOptions();
  renderPlayersInputs();
</script>
</body>
</html>
